\begin{frame}{Coefficient matrix}
	$$	\left[\begin{array}{rr}
				3 & 1 \\
				0 & 2 \\
			\end{array}\right]
		\left[\begin{array}{r}
				x \\
				y \\
			\end{array}\right]
		=\left[\begin{array}{r}
				\cdots \\
				\cdots \\
			\end{array}\right]
		or
		\left[\begin{array}{rr}
				3 & 1 \\
				0 & 2 \\
			\end{array}\right]
		\left[\begin{array}{r}
				a \\
				b \\
			\end{array}\right]
		=\left[\begin{array}{r}
				\cdots \\
				\cdots \\
			\end{array}\right]
	$$
	Coefficient matrix
	$	A = 	\left[\begin{array}{rr}
				3 & 1 \\
				0 & 2 \\
			\end{array}\right]
	$
	is also a rectangular matrix.\\[0.2cm]
	$det(A) = \left|\begin{array}{rr}
			3 & 1 \\
			0 & 2 \\
		\end{array}\right| =6$\\[0.2cm]
	Obvious matrix A has two vectors:$		\left[\begin{array}{r}
				3 \\
				0 \\
			\end{array}\right]
	$,
	$		\left[\begin{array}{r}
				1 \\
				2 \\
			\end{array}\right]
	$
\end{frame}
